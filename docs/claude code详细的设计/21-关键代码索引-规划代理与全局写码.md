## 1. 规划入口与退出

### `tools/EnterPlanModeTool/EnterPlanModeTool.ts`

- 文件地址：`tools/EnterPlanModeTool/EnterPlanModeTool.ts`
- 关键函数：`EnterPlanModeTool.call()`、`prompt()`、`mapToolResultToToolResultBlockParam(...)`
- 为什么关键：Plan Mode 不是提示词话术，而是权限模式和运行态切换。
- 关键伪代码：

```ts
const EnterPlanModeTool = buildTool({
  shouldDefer: true,
  isReadOnly: true,

  async call(_input, context) {
    if (context.agentId) throw new Error('no plan mode in subagent context')

    appState = context.getAppState()
    handlePlanModeTransition(appState.toolPermissionContext.mode, 'plan')

    context.setAppState(prev => ({
      ...prev,
      toolPermissionContext: applyPermissionUpdate(

        { type: 'setMode', mode: 'plan', destination: 'session' },
      ),
    }))

    return { message: 'Entered plan mode' }
  }
})
```

### `tools/ExitPlanModeTool/ExitPlanModeV2Tool.ts`

- 文件地址：`tools/ExitPlanModeTool/ExitPlanModeV2Tool.ts`
- 关键函数：`validateInput()`、`checkPermissions()`、`call()`
- 为什么关键：它把“从规划到执行”的切换变成受控状态迁移，而不是一句“开始做吧”。
- 关键伪代码：

```ts
async function call(input, context) {
  filePath = getPlanFilePath(context.agentId)
  inputPlan = input.plan ?? undefined
  plan = inputPlan ?? getPlan(context.agentId)

  if (inputPlan exists) {
    writeFile(filePath, inputPlan)
    persistFileSnapshotIfRemote()
  }

  if (isTeammate() && isPlanModeRequired()) {
    requestId = generateRequestId(...)
    writeToMailbox('team-lead', {
      type: 'plan_approval_request',
      planFilePath: filePath,
      planContent: plan,
      requestId,
    })
    setAwaitingPlanApproval(...)
    return { awaitingLeaderApproval: true, requestId, plan }
  }

  context.setAppState(prev => {
    restoreMode = prev.toolPermissionContext.prePlanMode ?? 'default'
    if (restoreMode === 'auto' && !autoModeGateEnabled()) {
      restoreMode = 'default'
    }

    if (restoreMode === 'auto') {
      baseContext = stripDangerousPermissionsForAutoMode(prev.toolPermissionContext)
    } else if (prev.toolPermissionContext.strippedDangerousRules) {
      baseContext = restoreDangerousPermissions(prev.toolPermissionContext)
    }

    return {
      ...prev,
      toolPermissionContext: {
        ...baseContext,
        mode: restoreMode,
        prePlanMode: undefined,
      },
    }
  })

  return { plan, filePath, hasTaskTool }
}
```

## 2. 计划文件系统

### `utils/plans.ts`

- 文件地址：`utils/plans.ts`
- 关键函数：`getPlansDirectory()`、`getPlanFilePath()`、`getPlan()`、`copyPlanForResume()`、`copyPlanForFork()`、`recoverPlanFromMessages()`
- 为什么关键：计划是磁盘工件，不只是会话中的一段文本。
- 关键伪代码：

```ts
function getPlansDirectory() {
  settingsDir = settings.plansDirectory
  resolved = resolve(cwd, settingsDir)
  if (resolved escapes project root) {
    plansPath = fallbackConfigHomePlansDir()
  } else {
    plansPath = resolved
  }
  mkdirSync(plansPath)
  return plansPath
}

function getPlanFilePath(agentId?) {
  slug = getPlanSlug(getSessionId())
  return !agentId
    ? join(getPlansDirectory(), `${slug}.md`)
    : join(getPlansDirectory(), `${slug}-agent-${agentId}.md`)
}

async function copyPlanForResume(log, targetSessionId) {
  slug = getSlugFromLog(log)
  if (!slug) return false

  setPlanSlug(targetSessionId, slug)
  if (planFileExists(slug)) return true

  recovered = findFileSnapshotEntry(log.messages, 'plan')
    ?? recoverPlanFromMessages(log)

  if (recovered) {
    await writeFile(planPath, recovered)
    return true
  }
  return false
}

function recoverPlanFromMessages(log) {
  scan messages from end to start:
    1. ExitPlanMode tool_use input.plan
    2. user.planContent
    3. plan_file_reference attachment
}
```

async function call({ todos }, context) {
  todoKey = context.agentId ?? getSessionId()
  oldTodos = appState.todos[todoKey] ?? []

### `tools/TaskUpdateTool/TaskUpdateTool.ts`

- 文件地址：`tools/TaskUpdateTool/TaskUpdateTool.ts`
- 关键函数：`TaskUpdateTool.call()`、`mapToolResultToToolResultBlockParam(...)`
- 为什么关键：这是正式任务图谱的更新器，支持 owner、status、metadata、blocks、blockedBy。
- 关键伪代码：

```ts
async function call({ taskId, subject, description, status, owner, addBlocks, addBlockedBy, metadata }, context) {
  existingTask = await getTask(taskListId, taskId)
  if (!existingTask) return failure('Task not found')

  updates = {}
  if (subject changed) updates.subject = subject
  if (description changed) updates.description = description
  if (owner changed) updates.owner = owner
  if (status === 'in_progress' && owner missing && teammate) {
    updates.owner = getAgentName()
  }
  if (metadata provided) updates.metadata = mergeMetadata(existingTask.metadata, metadata)

  if (status === 'completed') {
    await executeTaskCompletedHooks(...)
  }
  if (status changed) updates.status = status

  await updateTask(taskListId, taskId, updates)

  if (updates.owner) {
    writeToMailbox(updates.owner, assignmentMessage)
  }

  for (blockId of addBlocks) await blockTask(taskListId, taskId, blockId)
  for (blockerId of addBlockedBy) await blockTask(taskListId, blockerId, taskId)

  verificationNudgeNeeded =
    mainThread && allTasksDone && taskCount >= 3 && noVerificationTaskPresent

  return { success: true, updatedFields, statusChange, verificationNudgeNeeded }
}
```

## 4. 子代理上下文与 fork

### `utils/forkedAgent.ts`

- 文件地址：`utils/forkedAgent.ts`
- 关键函数：`createSubagentContext()`、`runForkedAgent()`
- 为什么关键：子代理不是裸 prompt，而是克制地继承父上下文的受控执行现场。
- 关键伪代码：

```ts
function createSubagentContext(parentContext, overrides?) {
  abortController =
    overrides.abortController
    ?? (overrides.shareAbortController
        ? parentContext.abortController
        : createChildAbortController(parentContext.abortController))

  getAppState =
    overrides.getAppState
    ?? (overrides.shareAbortController
        ? parentContext.getAppState
        : wrapGetAppStateToSetShouldAvoidPermissionPrompts(parentContext))

  return {
    readFileState: cloneFileStateCache(overrides.readFileState ?? parentContext.readFileState),
    nestedMemoryAttachmentTriggers: new Set(),
    loadedNestedMemoryPaths: new Set(),
    dynamicSkillDirTriggers: new Set(),
    discoveredSkillNames: new Set(),
    contentReplacementState: cloneOrOverride(parentContext.contentReplacementState),
    getAppState,
    setAppState: overrides.shareSetAppState ? parentContext.setAppState : noOp,
    options: overrides.options ?? parentContext.options,
    messages: overrides.messages ?? parentContext.messages,
    agentId: overrides.agentId ?? createAgentId(),
    queryTracking: newSubagentQueryTracking(parentContext.queryTracking),
  }
}

async function runForkedAgent({ promptMessages, cacheSafeParams, canUseTool, querySource, forkLabel, overrides }) {
  isolatedContext = createSubagentContext(cacheSafeParams.toolUseContext, overrides)
  initialMessages = [...cacheSafeParams.forkContextMessages, ...promptMessages]

  for await (message of query({
    messages: initialMessages,
    systemPrompt,
    userContext,
    systemContext,
    canUseTool,
    toolUseContext: isolatedContext,
    querySource,
  })) {
    collectOutput(message)
  }

  return { outputMessages, totalUsage }
}
```

## 5. 全局写码入口

### `tools/FileReadTool/FileReadTool.ts`

- 文件地址：`tools/FileReadTool/FileReadTool.ts`
- 关键函数：`preparePermissionMatcher(...)`、`checkPermissions(...)`、`validateInput(...)`、`call(...)`
- 为什么关键：读文件不仅是拿内容，还会把文件事实写进 `readFileState`，为后续安全编辑提供依据。
- 关键伪代码：

```ts
const FileReadTool = buildTool({
  searchHint: 'read files, images, PDFs, notebooks',
  isReadOnly: true,

  async validateInput({ file_path, pages }, context) {
    validatePdfPageRangeIfNeeded()
    fullFilePath = expandPath(file_path)

    if (denyRuleMatches(fullFilePath)) reject()
    if (isUNCPath(fullFilePath)) allowEarly()
    if (binaryExtensionButNotImageOrPdf(fullFilePath)) reject()
    if (isBlockedDevicePath(fullFilePath)) reject()

    return allow()
  }

  async call({ file_path, offset, limit, pages }, context) {
    fullFilePath = expandPath(file_path)

    if (sameRangeAlreadyReadAndFileUnchanged(fullFilePath, offset, limit)) {
      return file_unchanged_stub()
    }

    result = await callInner(...)
    context.readFileState.set(fullFilePath, {
      timestamp,
      content,
      offset,
      limit,
      isPartialView,
    })
    return result
  }
})
```

### `tools/FileEditTool/FileEditTool.ts`

- 文件地址：`tools/FileEditTool/FileEditTool.ts`
- 关键函数：`checkPermissions(...)`、`validateInput(...)`、`call(...)`
- 为什么关键：真正决定系统什么时候允许写代码。它是“基于事实写码”的硬约束层。
- 关键伪代码：

```ts
async function validateInput({ file_path, old_string, new_string, replace_all }, context) {
  fullFilePath = expandPath(file_path)

  if (old_string === new_string) reject()
  if (denyRuleMatches(fullFilePath)) reject()
  if (isUNCPath(fullFilePath)) allowEarly()
  if (fileTooLarge(fullFilePath)) reject()

  fileContent = readFileIfExists(fullFilePath)
  if (file missing && old_string === '') allowCreate()
  if (file exists && old_string === '' && file not empty) reject()
  if (isNotebook(fullFilePath)) reject()

  readTimestamp = context.readFileState.get(fullFilePath)
  if (!readTimestamp || readTimestamp.isPartialView) reject('read first')

  if (file modified since read and content differs) reject('read again first')
  if (!findActualString(fileContent, old_string)) reject('string not found')
  if (multipleMatches && !replace_all) reject('need more context or replace_all')

  return allow()
}
```

### `tools/LSPTool/LSPTool.ts`

- 文件地址：`tools/LSPTool/LSPTool.ts`
- 关键函数：`validateInput(...)`、`checkPermissions(...)`、`call(...)`、`getMethodAndParams(...)`、`formatResult(...)`
- 为什么关键：系统不只是读文件文本，还会用 LSP 读程序结构。
- 关键伪代码：

```ts
const LSPTool = buildTool({
  searchHint: 'code intelligence (definitions, references, symbols, hover)',
  isReadOnly: true,
  shouldDefer: true,

  async call(input) {
    absolutePath = expandPath(input.filePath)
    await waitForInitializationIfPending()
    manager = getLspServerManager()
    if (!manager) return noManagerError()

    { method, params } = getMethodAndParams(input, absolutePath)

    if (!manager.isFileOpen(absolutePath)) {
      fileContent = await readFile(absolutePath)
      await manager.openFile(absolutePath, fileContent)
    }

    result = await manager.sendRequest(absolutePath, method, params)

    if (input.operation is incomingCalls/outgoingCalls) {
      prepared = result[0]
      result = await manager.sendRequest(absolutePath, callHierarchyMethod, { item: prepared })
    }

    result = maybeFilterGitIgnoredLocations(result)
    return formatResult(input.operation, result, cwd)
  }
})
```

### `services/tools/toolOrchestration.ts`

- 文件地址：`services/tools/toolOrchestration.ts`
- 关键函数：`runTools()`、`partitionToolCalls()`、`runToolsSerially()`、`runToolsConcurrently()`
- 为什么关键：当不是流式工具执行时，这里负责把工具按并发安全性分组成 batch。
- 关键伪代码：

```ts
async function* runTools(toolUseMessages, assistantMessages, canUseTool, context) {
  currentContext = context
  for (batch of partitionToolCalls(toolUseMessages, currentContext)) {
    if (batch.isConcurrencySafe) {
      updates = runToolsConcurrently(batch.blocks, ...)
      queueContextModifiersThenApplyInOrder()
    } else {
      updates = runToolsSerially(batch.blocks, ...)
      applyContextModifiersInline()
    }
    yield updates
  }
}

function partitionToolCalls(toolUseMessages, context) {
  group consecutive concurrency-safe tools together
  keep each non-safe tool as exclusive batch
}
```
