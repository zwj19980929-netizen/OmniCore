# P2-1: 多模态输入/输出扩展方案

> 优先级：P2 | 预估工作量：3-5 天 | 依赖：无

---

## 1. 目标

扩展 OmniCore 的输入/输出能力，支持图片、文档、语音等多模态形式，让用户能以更自然的方式下发任务并接收结果。

**验收标准**：
- CLI 和 Streamlit UI 支持粘贴/上传图片后直接提问
- 支持上传 PDF/DOCX 后提取内容并作为任务上下文
- 语音输入（Whisper）：说话即任务
- 语音输出（TTS）：任务结果朗读（可选）

---

## 4. 详细设计

### 4.1 多模态输入处理器

新建 `utils/multimodal_input.py`：

```python
"""
MultimodalInputProcessor — 将多模态输入统一转换为文本 user_input。

处理顺序：
1. 语音文件 → Whisper → 文本
2. 图片 → Vision LLM → 意图描述
3. 文档（PDF/DOCX）→ DocumentParser → 文本摘要
4. 组合 → 合并为最终 user_input
"""

import os
from dataclasses import dataclass, field
from pathlib import Path
from typing import List, Optional, Tuple

from core.llm import LLMClient
from utils.logger import log_agent_action, log_warning


@dataclass
class MultimodalInput:
    """多模态输入的结构化表示。"""
    text: str = ""                          # 文字部分
    image_paths: List[str] = field(default_factory=list)   # 图片路径列表
    audio_path: str = ""                    # 语音文件路径
    document_paths: List[str] = field(default_factory=list) # 文档路径列表


class MultimodalInputProcessor:
    """将多模态输入转换为标准文本 user_input。"""

    def __init__(self):
        self._vision_llm: Optional[LLMClient] = None
        self._asr_llm: Optional[LLMClient] = None

    def _get_vision_llm(self) -> LLMClient:
        if self._vision_llm is None:
            self._vision_llm = LLMClient.for_vision()
        return self._vision_llm

    def _get_asr_llm(self) -> LLMClient:
        if self._asr_llm is None:
            self._asr_llm = LLMClient()
        return self._asr_llm

    async def process(self, inp: MultimodalInput) -> str:
        """
        将多模态输入处理为最终 user_input 字符串。

        处理策略：
        - 只有语音：转文字后作为 user_input
        - 只有图片 + 文字：用 Vision 描述图片，追加到文字后
        - 只有图片（无文字）：Vision 推断意图
        - 文档：提取文本摘要，追加为上下文
        - 混合：逐步处理后合并
        """
        parts = []

        # 1. 处理语音输入
        if inp.audio_path:
            transcript = await self._process_audio(inp.audio_path)
            if transcript:
                parts.append(transcript)

        # 2. 处理文字部分（合并语音识别结果）
        if inp.text:
            parts.append(inp.text)

        # 3. 处理图片
        if inp.image_paths:
            image_context = await self._process_images(
                inp.image_paths,
                hint=" ".join(parts) if parts else "",
            )
            if image_context:
                parts.append(f"\n[图片内容]\n{image_context}")

        # 4. 处理文档
        if inp.document_paths:
            doc_context = await self._process_documents(inp.document_paths)
            if doc_context:
                parts.append(f"\n[文档内容]\n{doc_context}")

        return "\n".join(p for p in parts if p).strip()

    async def _process_audio(self, audio_path: str) -> str:
        """语音转文字（Whisper）。"""
        try:
            llm = self._get_asr_llm()
            transcript = llm.transcribe(audio_path, language="zh")
            log_agent_action("MultimodalInput", f"Audio transcribed: {len(transcript)} chars")
            return transcript
        except Exception as e:
            log_warning(f"Audio transcription failed: {e}")
            return ""

    async def _process_images(self, image_paths: List[str], hint: str = "") -> str:
        """图片 → Vision LLM → 描述/意图。"""
        descriptions = []
        llm = self._get_vision_llm()

        prompt = (
            f"用户意图提示：{hint}\n\n" if hint else ""
        ) + (
            "请描述图片内容，并推断用户可能想用这张图片完成什么任务。"
            "如果图片包含文字，请完整提取。如果是截图，描述界面内容。"
            "回答简洁，100字以内。"
        )

        for path in image_paths[:3]:  # 最多处理 3 张
            try:
                response = llm.chat_with_image(text=prompt, image=path)
                descriptions.append(response.content)
                log_agent_action("MultimodalInput", f"Image processed: {path}")
            except Exception as e:
                log_warning(f"Image processing failed for {path}: {e}")

        return "\n".join(descriptions)

    async def _process_documents(self, doc_paths: List[str]) -> str:
        """文档提取 + 摘要。"""
        from utils.document_parser import extract_text  # 来自 P1-2 的文档解析器

        summaries = []
        llm = LLMClient()

        for path in doc_paths[:3]:  # 最多 3 个文档
            try:
                content = extract_text(path)
                if not content:
                    continue

                # 长文档先截断再摘要
                if len(content) > 8000:
                    prompt = f"请用200字以内摘要以下文档内容：\n\n{content[:8000]}"
                    response = llm.chat_with_system("你是文档摘要助手。", prompt)
                    summary = response.content
                else:
                    summary = content[:2000]

                filename = Path(path).name
                summaries.append(f"文件：{filename}\n{summary}")
                log_agent_action("MultimodalInput", f"Document processed: {filename}")

            except Exception as e:
                log_warning(f"Document processing failed for {path}: {e}")

        return "\n\n".join(summaries)
```

### 4.2 CLI 多模态支持

在 `main.py` 中新增 `/attach` 命令：

```python
elif user_input.startswith("/attach"):
    # /attach <file_path> [可选文字描述]
    parts = user_input[7:].strip().split(None, 1)
    if not parts:
        print("用法: /attach <file_path> [描述]")
        continue

    file_path = os.path.expanduser(parts[0])
    text_hint = parts[1] if len(parts) > 1 else ""

    if not os.path.exists(file_path):
        print(f"文件不存在: {file_path}")
        continue

    from utils.multimodal_input import MultimodalInputProcessor, MultimodalInput

    ext = os.path.splitext(file_path)[1].lower()
    processor = MultimodalInputProcessor()

    if ext in (".png", ".jpg", ".jpeg", ".webp", ".gif"):
        inp = MultimodalInput(text=text_hint, image_paths=[file_path])
    elif ext in (".mp3", ".wav", ".m4a", ".ogg"):
        inp = MultimodalInput(audio_path=file_path)
    elif ext in (".pdf", ".docx", ".txt", ".md"):
        inp = MultimodalInput(text=text_hint, document_paths=[file_path])
    else:
        print(f"不支持的文件类型: {ext}")
        continue

    # 处理后送入正常任务流程
    processed_input = await processor.process(inp)
    print(f"  已解析: {processed_input[:100]}...")
    # 作为下一次输入
    pending_input = processed_input
```

也可以支持直接粘贴文件路径：

```python
# 自动检测用户输入是否包含有效文件路径
def _try_detect_file_input(user_input: str) -> Optional[str]:
    """如果用户输入像文件路径且文件存在，返回处理后的多模态输入。"""
    stripped = user_input.strip()
    expanded = os.path.expanduser(stripped.split()[0] if stripped else "")
    if os.path.isfile(expanded):
        ext = os.path.splitext(expanded)[1].lower()
        if ext in (".png", ".jpg", ".jpeg", ".pdf", ".docx", ".mp3", ".wav"):
            return expanded
    return None
```

        from utils.multimodal_input import MultimodalInputProcessor, MultimodalInput

        file_path = ""
        if uploaded_file:
            # 临时保存上传文件
            with tempfile.NamedTemporaryFile(
                suffix=os.path.splitext(uploaded_file.name)[1],
                delete=False,
            ) as tmp:
                tmp.write(uploaded_file.read())
                file_path = tmp.name

        processor = MultimodalInputProcessor()

        if ext in (".png", ".jpg", ".jpeg"):
            inp = MultimodalInput(text=user_text, image_paths=[file_path])
        elif ext in (".mp3", ".wav", ".m4a"):
            inp = MultimodalInput(audio_path=file_path)
        elif file_path:
            inp = MultimodalInput(text=user_text, document_paths=[file_path])
        else:
            inp = MultimodalInput(text=user_text)

        final_input = asyncio.run(processor.process(inp))
        # 走正常任务提交
        result = run_task(final_input, ...)

## 6. 风险与注意事项

| 风险 | 缓解措施 |
|------|----------|
| Vision 调用成本高 | 图片输入走 Vision budget 限制（复用现有 VisionBudget） |
| 大文件处理慢 | 文档截断 8000 字符；语音文件限制 25MB（Whisper 限制）|
| 临时文件泄露 | 处理后立即删除临时文件（atexit 注册清理） |
| 语音输入在嘈杂环境失准 | 语音识别结果显示给用户确认后再执行 |
| Streamlit 上传大文件超时 | 设置 `client.max_upload_size = 200` MB |
