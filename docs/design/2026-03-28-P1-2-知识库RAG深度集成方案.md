# P1-2: 知识库 & RAG 深度集成方案

> 优先级：P1 | 预估工作量：3-4 天 | 依赖：无

---

## 1. 目标

将 OmniCore 的 ChromaDB 记忆系统升级为完整的个人知识库，实现：自动索引（浏览产出、下载文件、任务结果）+ 任务时自动检索注入上下文（RAG）+ 用户主动学习（`/learn`）。

**验收标准**：
- 浏览器抓取的网页内容、下载的文件自动索引到知识库
- Router/Planner 阶段自动检索相关知识并注入 LLM 上下文
- `/learn <url_or_file>` 命令主动添加知识
- `/knowledge search <query>` 命令查询知识库

## 4. 详细设计

### 4.1 知识库存储层

新建 `memory/knowledge_store.py`：

```python
"""
KnowledgeStore — 个人知识库的存储与检索。

基于 ChromaDB，按内容类型分 collection：
- web_pages: 浏览器/爬虫抓取的网页内容
- documents: 用户主动导入的文档（PDF/TXT/MD/DOCX）
- task_results: 任务执行的关键结果摘要
"""

import hashlib
import json
from dataclasses import dataclass, field
from datetime import datetime
from typing import Any, Dict, List, Optional

from memory.scoped_chroma_store import ScopedChromaStore
from utils.logger import log_agent_action, log_warning


@dataclass
class KnowledgeItem:
    """知识条目。"""
    content: str                            # 文本内容
    source_type: str                        # web_page | document | task_result
    source_url: str = ""                    # 来源 URL 或文件路径
    title: str = ""                         # 标题
    tags: List[str] = field(default_factory=list)
    metadata: Dict[str, Any] = field(default_factory=dict)


class KnowledgeStore:
    """个人知识库管理。"""

    # 每个 collection 的文本分块大小
    CHUNK_SIZE = 1000           # 字符
    CHUNK_OVERLAP = 200         # 重叠字符

    def __init__(self):
        self._chroma = ScopedChromaStore()

    # ── 索引 ──────────────────────────────────

    async def index_web_page(
        self,
        url: str,
        title: str,
        content: str,
        session_id: str = "",
        job_id: str = "",
    ) -> int:
        """索引网页内容。返回索引的 chunk 数。"""
        chunks = self._split_text(content)
        if not chunks:
            return 0

        fingerprint = hashlib.md5(url.encode()).hexdigest()[:12]

        for i, chunk in enumerate(chunks):
            doc_id = f"web_{fingerprint}_{i}"
            self._chroma.upsert(
                document=chunk,
                doc_id=doc_id,
                memory_type="web_page",
                metadata={
                    "source_url": url,
                    "title": title,
                    "chunk_index": str(i),
                    "total_chunks": str(len(chunks)),
                    "session_id": session_id,
                    "job_id": job_id,
                    "indexed_at": datetime.now().isoformat(),
                },
            )

        log_agent_action("KnowledgeStore", f"Indexed web page: {title} ({len(chunks)} chunks)")
        return len(chunks)

    async def index_document(
        self,
        file_path: str,
        content: str,
        title: str = "",
    ) -> int:
        """索引文档文件。返回索引的 chunk 数。"""
        chunks = self._split_text(content)
        if not chunks:
            return 0

        fingerprint = hashlib.md5(file_path.encode()).hexdigest()[:12]
        if not title:
            title = file_path.split("/")[-1]

        for i, chunk in enumerate(chunks):
            doc_id = f"doc_{fingerprint}_{i}"
            self._chroma.upsert(
                document=chunk,
                doc_id=doc_id,
                memory_type="document",
                metadata={
                    "source_path": file_path,
                    "title": title,
                    "chunk_index": str(i),
                    "total_chunks": str(len(chunks)),
                    "indexed_at": datetime.now().isoformat(),
                },
            )

        log_agent_action("KnowledgeStore", f"Indexed document: {title} ({len(chunks)} chunks)")
        return len(chunks)

    async def index_task_result(
        self,
        summary: str,
        user_input: str,
        job_id: str,
        session_id: str = "",
    ) -> None:
        """索引任务结果摘要。"""
        if len(summary) < 50:  # 太短的结果不索引
            return

        doc_id = f"task_{job_id}"
        self._chroma.upsert(
            document=summary,
            doc_id=doc_id,
            memory_type="task_result",
            metadata={
                "user_input": user_input[:200],
                "job_id": job_id,
                "session_id": session_id,
                "indexed_at": datetime.now().isoformat(),
            },
        )

    # ── 检索（RAG） ──────────────────────────

    async def retrieve(
        self,
        query: str,
        top_k: int = 5,
        source_types: Optional[List[str]] = None,
        max_total_chars: int = 4000,
    ) -> List[Dict[str, Any]]:
        """
        语义检索相关知识。

        Args:
            query: 检索查询
            top_k: 每个 collection 的返回数量
            source_types: 限定来源类型，None 表示全部
            max_total_chars: 返回文本总长度上限

        Returns:
            [{"content": str, "source": str, "title": str, "distance": float}, ...]
        """
        types_to_search = source_types or ["web_page", "document", "task_result"]
        all_results = []

        for memory_type in types_to_search:
            try:
                results = self._chroma.query(
                    query=query,
                    n_results=top_k,
                    memory_type=memory_type,
                )
                if results and results.get("documents"):
                    for doc, meta, dist in zip(
                        results["documents"][0],
                        results["metadatas"][0],
                        results["distances"][0],
                    ):
                        if dist > 0.5:  # 过滤低相关性结果
                            continue
                        all_results.append({
                            "content": doc,
                            "source": meta.get("source_url") or meta.get("source_path") or meta.get("job_id", ""),
                            "title": meta.get("title", ""),
                            "distance": dist,
                            "type": memory_type,
                        })
            except Exception as e:
                log_warning(f"Knowledge retrieval failed for {memory_type}: {e}")

        # 按相关性排序
        all_results.sort(key=lambda x: x["distance"])

        # 截断到 max_total_chars
        selected = []
        total_chars = 0
        for r in all_results:
            if total_chars + len(r["content"]) > max_total_chars:
                break
            selected.append(r)
            total_chars += len(r["content"])

        return selected

    def format_as_context(self, results: List[Dict]) -> str:
        """将检索结果格式化为 LLM 上下文注入。"""
        if not results:
            return ""

        lines = ["## 相关知识（来自知识库）\n"]
        for r in results:
            source_label = r.get("title") or r.get("source") or "unknown"
            lines.append(f"### [{r['type']}] {source_label}")
            lines.append(r["content"])
            lines.append("")

        return "\n".join(lines)

    # ── 管理 ──────────────────────────────────

    def search(self, query: str, limit: int = 10) -> List[Dict]:
        """CLI 查询接口。"""
        import asyncio
        loop = asyncio.get_event_loop()
        return loop.run_until_complete(self.retrieve(query, top_k=limit))

    def delete_by_source(self, source: str) -> int:
        """按来源删除知识条目。"""
        # 通过 metadata 过滤删除
        count = 0
        for memory_type in ["web_page", "document", "task_result"]:
            try:
                results = self._chroma.query(
                    query=source,
                    n_results=100,
                    memory_type=memory_type,
                    where={"source_url": source},
                )
                if results and results.get("ids"):
                    ids = results["ids"][0]
                    if ids:
                        self._chroma.delete(ids=ids, memory_type=memory_type)
                        count += len(ids)
            except Exception:
                pass
        return count

    def get_stats(self) -> Dict[str, int]:
        """返回各 collection 的条目数。"""
        stats = {}
        for memory_type in ["web_page", "document", "task_result"]:
            try:
                count = self._chroma.count(memory_type=memory_type)
                stats[memory_type] = count
            except Exception:
                stats[memory_type] = 0
        return stats

    # ── 内部 ──────────────────────────────────

    def _split_text(self, text: str) -> List[str]:
        """将长文本分块。"""
        if not text or len(text) < 100:
            return [text] if text else []

        chunks = []
        start = 0
        while start < len(text):
            end = start + self.CHUNK_SIZE
            chunk = text[start:end]
            if chunk.strip():
                chunks.append(chunk.strip())
            start = end - self.CHUNK_OVERLAP

        return chunks
```

### 4.4 文档解析器

新建 `utils/document_parser.py`：

```python
"""文档内容提取 — 支持 PDF / TXT / MD / DOCX / HTML / CSV。"""

import os
from typing import Optional


def extract_text(file_path: str) -> Optional[str]:
    """从文件提取纯文本内容。"""
    ext = os.path.splitext(file_path)[1].lower()

    if ext in (".txt", ".md", ".py", ".js", ".json", ".yaml", ".yml", ".csv"):
        with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
            return f.read()

    elif ext == ".pdf":
        return _extract_pdf(file_path)

    elif ext in (".html", ".htm"):
        return _extract_html(file_path)

    elif ext == ".docx":
        return _extract_docx(file_path)

    return None


def _extract_pdf(path: str) -> Optional[str]:
    try:
        import PyPDF2
        with open(path, "rb") as f:
            reader = PyPDF2.PdfReader(f)
            pages = [page.extract_text() or "" for page in reader.pages]
            return "\n\n".join(pages)
    except ImportError:
        return None


def _extract_html(path: str) -> Optional[str]:
    try:
        from bs4 import BeautifulSoup
        with open(path, "r", encoding="utf-8", errors="ignore") as f:
            soup = BeautifulSoup(f.read(), "html.parser")
            return soup.get_text(separator="\n", strip=True)
    except ImportError:
        return None


def _extract_docx(path: str) -> Optional[str]:
    try:
        import docx
        doc = docx.Document(path)
        return "\n\n".join(p.text for p in doc.paragraphs if p.text.strip())
    except ImportError:
        return None
```

## 7. 风险与注意事项

| 风险 | 缓解措施 |
|------|----------|
| 知识库无限膨胀 | 定期清理策略：`indexed_at` 超过 90 天的低使用条目自动归档 |
| 检索噪音注入 LLM | distance > 0.5 过滤 + 总长度限制 4000 字符 |
| 文档解析依赖缺失 | PyPDF2 / python-docx 设为可选依赖，缺失时跳过 |
| ChromaDB 性能 | 分 collection 存储；个人使用数万条无压力 |
| 敏感信息索引 | 不索引含密钥/密码的内容；提供 `/knowledge delete` 手动清理 |
