# P2-2: 成本感知与智能路由方案

> 优先级：P2 | 预估工作量：2-3 天 | 依赖：无

---

## 1. 目标

基于任务复杂度自动选择最优性价比模型：简单任务用便宜模型（DeepSeek/GPT-4o-mini），复杂推理用强模型（GPT-4o/Claude）。提供成本统计命令，支持月度预算上限。

**验收标准**：
- Router 阶段自动为任务打 complexity score（0~1.0）
- LLMClient 根据 complexity 动态选择模型
- `/cost` 命令查看历史成本统计
- 设置 `MONTHLY_BUDGET_USD` 后接近上限时发出警告

### 4.1 复杂度评分器

新建 `core/complexity_scorer.py`：

```python
"""
TaskComplexityScorer — 对 Router 输出的任务列表打复杂度分。

评分维度（0~1.0）：
- 步骤数（多步骤更复杂）
- 工具类型（Browser > System > Web > File）
- 任务描述长度和关键词
- 依赖链深度
- 历史失败率（可选，从 MetricsStore 获取）
"""

import re
from typing import Any, Dict, List

# 工具复杂度基准分
TOOL_COMPLEXITY = {
    "browser_agent": 0.8,
    "browser.": 0.8,
    "system_worker": 0.7,
    "terminal_worker": 0.7,
    "web_worker": 0.4,
    "web.": 0.4,
    "file_worker": 0.2,
    "file.": 0.2,
    "mcp.": 0.3,
}

# 高复杂度关键词
COMPLEX_KEYWORDS = [
    "分析", "对比", "总结", "规划", "设计", "生成报告", "汇总",
    "多步", "自动化", "批量", "爬取所有", "完整流程",
    "analyze", "compare", "summarize", "plan", "design", "automate",
]

# 低复杂度关键词
SIMPLE_KEYWORDS = [
    "查看", "打开", "搜索一下", "告诉我", "查询",
    "open", "show", "search", "tell me", "lookup",
]


def score_task_complexity(tasks: List[Dict[str, Any]], user_input: str = "") -> float:
    """
    对任务列表整体打复杂度分（0~1.0）。

    0.0~0.3: 简单（单步查询）→ 用 low cost 模型
    0.3~0.6: 中等（多步执行）→ 用 medium cost 模型
    0.6~1.0: 复杂（浏览器/推理/批量）→ 用 high cost 模型
    """
    if not tasks:
        return _score_from_input_only(user_input)

    scores = []

    # 1. 步骤数维度
    step_count = len(tasks)
    step_score = min(step_count / 5.0, 1.0)    # 5 步以上算满分
    scores.append(step_score * 0.25)

    # 2. 工具类型维度
    tool_max = 0.0
    for task in tasks:
        tool_name = (task.get("tool_name") or "").lower()
        for prefix, base_score in TOOL_COMPLEXITY.items():
            if tool_name.startswith(prefix) or prefix in tool_name:
                tool_max = max(tool_max, base_score)
                break
    scores.append(tool_max * 0.35)

    # 3. 依赖链深度
    max_depth = _compute_dependency_depth(tasks)
    depth_score = min(max_depth / 3.0, 1.0)
    scores.append(depth_score * 0.15)

    # 4. 用户输入关键词
    input_score = _score_from_input_only(user_input)
    scores.append(input_score * 0.25)

    total = sum(scores)
    return round(min(total, 1.0), 3)


def _score_from_input_only(user_input: str) -> float:
    """仅基于输入文本估算复杂度。"""
    if not user_input:
        return 0.3

    lowered = user_input.lower()

    # 关键词匹配
    complex_hits = sum(1 for kw in COMPLEX_KEYWORDS if kw in lowered)
    simple_hits = sum(1 for kw in SIMPLE_KEYWORDS if kw in lowered)

    # 长度维度（长输入通常更复杂）
    length_score = min(len(user_input) / 200.0, 0.5)

    base = 0.3 + length_score * 0.2
    base += complex_hits * 0.1
    base -= simple_hits * 0.1

    return round(max(0.1, min(base, 1.0)), 3)


def _compute_dependency_depth(tasks: List[Dict[str, Any]]) -> int:
    """计算任务依赖链的最大深度。"""
    id_to_idx = {t.get("task_id"): i for i, t in enumerate(tasks)}
    max_depth = 0

    def dfs(task_id: str, visited: set) -> int:
        if task_id in visited:
            return 0
        visited.add(task_id)
        idx = id_to_idx.get(task_id)
        if idx is None:
            return 0
        task = tasks[idx]
        deps = task.get("depends_on") or []
        if not deps:
            return 1
        return 1 + max(dfs(dep, visited.copy()) for dep in deps)

    for task in tasks:
        depth = dfs(task.get("task_id", ""), set())
        max_depth = max(max_depth, depth)

    return max_depth


def complexity_to_cost_preference(complexity: float) -> str:
    """将复杂度分映射到 cost_preference。"""
    if complexity < 0.35:
        return "low"
    elif complexity < 0.65:
        return "medium"
    else:
        return "high"
```

### 4.2 成本计算器

新建 `utils/cost_tracker.py`：

```python
"""
CostTracker — LLM 调用成本计算与统计。
"""

import json
import os
from datetime import datetime, timezone
from typing import Dict, List, Optional, Tuple

import yaml

from utils.logger import log_warning


class CostTracker:
    """计算和统计 LLM 调用成本。"""

    _pricing: Optional[Dict] = None

    @classmethod
    def _get_pricing(cls) -> Dict:
        if cls._pricing is None:
            pricing_path = os.path.join(
                os.path.dirname(__file__), "..", "config", "model_pricing.yaml"
            )
            if os.path.exists(pricing_path):
                with open(pricing_path, "r") as f:
                    cls._pricing = yaml.safe_load(f).get("pricing", {})
            else:
                cls._pricing = {}
        return cls._pricing

    @classmethod
    def calculate_cost(
        cls,
        model_full_name: str,
        tokens_in: int,
        tokens_out: int,
    ) -> float:
        """
        计算单次调用成本（USD）。

        model_full_name 格式：
        - "deepseek/deepseek-chat"
        - "openai/gpt-4o"
        - "gpt-4o"（不含 provider 前缀时自动查找）
        """
        pricing = cls._get_pricing()
        provider, model_id = cls._parse_model_name(model_full_name)

        if not provider:
            return 0.0

        provider_pricing = pricing.get(provider, {})
        model_pricing = provider_pricing.get(model_id, {})

        if not model_pricing:
            return 0.0

        input_cost = (tokens_in / 1_000_000) * model_pricing.get("input", 0)
        output_cost = (tokens_out / 1_000_000) * model_pricing.get("output", 0)
        return round(input_cost + output_cost, 8)

    @classmethod
    def _parse_model_name(cls, model_name: str) -> Tuple[str, str]:
        """解析 provider/model 格式。"""
        if "/" in model_name:
            parts = model_name.split("/", 1)
            return parts[0], parts[1]

        # 无 provider 前缀时按模型名猜测
        known_prefixes = {
            "gpt-": "openai",
            "claude-": "anthropic",
            "gemini-": "gemini",
            "deepseek-": "deepseek",
            "minimax-": "minimax",
            "MiniMax-": "minimax",
        }
        for prefix, provider in known_prefixes.items():
            if model_name.lower().startswith(prefix.lower()):
                return provider, model_name
        return "", model_name


class MonthlyCostGuard:
    """月度预算守卫。"""

    def __init__(self, monthly_budget_usd: float, data_dir: str = "data"):
        self.monthly_budget_usd = monthly_budget_usd
        self._cost_file = os.path.join(data_dir, "monthly_cost.jsonl")

    def record_cost(self, cost_usd: float, model: str, job_id: str = "") -> None:
        """记录一次调用成本。"""
        record = {
            "ts": datetime.now(timezone.utc).isoformat(),
            "cost_usd": cost_usd,
            "model": model,
            "job_id": job_id,
        }
        os.makedirs(os.path.dirname(self._cost_file), exist_ok=True)
        with open(self._cost_file, "a", encoding="utf-8") as f:
            f.write(json.dumps(record, ensure_ascii=False) + "\n")

    def get_current_month_cost(self) -> float:
        """统计当月累计成本。"""
        if not os.path.exists(self._cost_file):
            return 0.0

        now = datetime.now(timezone.utc)
        total = 0.0
        with open(self._cost_file, "r", encoding="utf-8") as f:
            for line in f:
                try:
                    rec = json.loads(line.strip())
                    ts = datetime.fromisoformat(rec["ts"])
                    if ts.year == now.year and ts.month == now.month:
                        total += rec.get("cost_usd", 0)
                except Exception:
                    continue
        return round(total, 6)

    def check_budget(self) -> Tuple[float, float, bool]:
        """
        检查预算状态。

        Returns:
            (used, budget, is_warning) — 超过 80% 时发出警告
        """
        used = self.get_current_month_cost()
        warning = used >= self.monthly_budget_usd * 0.8
        return used, self.monthly_budget_usd, warning

    def get_top_models_by_cost(self, limit: int = 10) -> List[Dict]:
        """按成本统计各模型消耗排名。"""
        if not os.path.exists(self._cost_file):
            return []

        from collections import defaultdict
        model_costs: Dict[str, float] = defaultdict(float)
        model_calls: Dict[str, int] = defaultdict(int)

        now = datetime.now(timezone.utc)
        with open(self._cost_file, "r", encoding="utf-8") as f:
            for line in f:
                try:
                    rec = json.loads(line.strip())
                    ts = datetime.fromisoformat(rec["ts"])
                    if ts.year == now.year and ts.month == now.month:
                        model = rec.get("model", "unknown")
                        model_costs[model] += rec.get("cost_usd", 0)
                        model_calls[model] += 1
                except Exception:
                    continue

        results = [
            {"model": m, "cost_usd": round(c, 6), "calls": model_calls[m]}
            for m, c in model_costs.items()
        ]
        results.sort(key=lambda x: x["cost_usd"], reverse=True)
        return results[:limit]
```

## 7. 风险与注意事项

| 风险 | 缓解措施 |
|------|----------|
| 复杂度打分不准 | 只影响模型选择，不影响执行；用户可通过 `COMPLEXITY_AWARE_ROUTING=false` 关闭 |
| 价格表过时 | `model_pricing.yaml` 定期手动更新；成本只是参考 |
| 预算守卫误报 | 只发 warning 日志，不阻断执行 |
| 升级到强模型后结果更慢 | 在成本和速度之间的权衡，由用户设置 `COST_PREFERENCE` 覆盖 |
