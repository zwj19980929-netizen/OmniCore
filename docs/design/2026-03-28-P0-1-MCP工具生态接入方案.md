# P0-1: MCP (Model Context Protocol) 工具生态接入方案

> 优先级：P0 | 预估工作量：2-3 天 | 依赖：无

---

## 1. 目标

通过接入 MCP 协议，让 OmniCore 能以声明式配置的方式对接社区 MCP Server（数据库查询、Slack/飞书/邮件、日历、GitHub API、文件系统等），无需为每个外部服务单独开发 Agent。

**验收标准**：
- 在 `config/mcp_servers.yaml` 配置一个 MCP Server 后，Router 自动感知其工具列表
- 用户下发任务时，Router 能选择 MCP 工具并正确执行
- MCP 工具的输入/输出与现有 TaskItem / ArtifactItem 体系无缝对接

---

## 2. 现有代码接入点分析

| 组件 | 文件 | 接入方式 |
|------|------|----------|
| 工具协议 | `core/tool_protocol.py` | MCP tool schema → `ToolSpec` 映射 |
| 工具注册 | `core/tool_registry.py` | 启动时从 MCP Server 拉取工具列表 → 动态 `register()` |
| 适配器 | `core/tool_adapters.py` | 新增 `MCPToolAdapter(BaseToolAdapter)` |
| Worker 池 | `core/tool_adapters.py` WorkerPool | 新增 `MCPClient` 作为 worker 实例 |
| Agent 注册 | `config/agents.yaml` | 新增 `mcp_handler` 条目 |
| Router 感知 | `core/router.py` `_build_router_system_prompt()` | 自动通过 `build_dynamic_tool_prompt_lines()` 注入 |
| 任务执行 | `core/task_executor.py` | 无需修改，`execute_tool_via_adapter()` 透明分发 |

### 3.2 MCP Client 实现

新建 `core/mcp_client.py`：

```python
"""
MCP Client — 管理与 MCP Server 的连接和工具调用。
支持 stdio / SSE / Streamable HTTP 三种传输方式。
"""

import asyncio
import json
import os
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

from utils.logger import log_agent_action, log_error, log_warning


@dataclass
class MCPServerConfig:
    """单个 MCP Server 的配置。"""
    name: str
    transport: str                          # stdio | sse | streamable_http
    enabled: bool = True
    description: str = ""
    description_zh: str = ""
    risk_level: str = "medium"
    max_parallelism: int = 2
    # stdio 专用
    command: str = ""
    args: List[str] = field(default_factory=list)
    env: Dict[str, str] = field(default_factory=dict)
    # SSE / HTTP 专用
    url: str = ""


@dataclass
class MCPTool:
    """从 MCP Server 发现的单个工具。"""
    server_name: str
    name: str                               # MCP tool name
    description: str
    input_schema: Dict[str, Any]            # JSON Schema


class MCPClient:
    """
    管理单个 MCP Server 的生命周期。

    职责：
    1. 启动/连接 Server
    2. 发现工具列表 (tools/list)
    3. 调用工具 (tools/call)
    4. 优雅关闭
    """

    def __init__(self, config: MCPServerConfig):
        self.config = config
        self._process: Optional[asyncio.subprocess.Process] = None
        self._tools: List[MCPTool] = []
        self._request_id: int = 0
        self._initialized: bool = False

    # ── 生命周期 ──────────────────────────────────

    async def connect(self) -> None:
        """建立与 MCP Server 的连接。"""
        if self.config.transport == "stdio":
            await self._connect_stdio()
        elif self.config.transport == "sse":
            await self._connect_sse()
        elif self.config.transport == "streamable_http":
            await self._connect_http()
        else:
            raise ValueError(f"Unsupported transport: {self.config.transport}")

        # 发送 initialize 请求
        await self._initialize_handshake()
        self._initialized = True

    async def disconnect(self) -> None:
        """优雅关闭连接。"""
        if self._process and self._process.returncode is None:
            self._process.terminate()
            try:
                await asyncio.wait_for(self._process.wait(), timeout=5.0)
            except asyncio.TimeoutError:
                self._process.kill()
        self._initialized = False

    # ── stdio 传输 ────────────────────────────────

    async def _connect_stdio(self) -> None:
        """启动子进程，通过 stdin/stdout 通信。"""
        env = os.environ.copy()
        for k, v in self.config.env.items():
            # 支持 ${ENV_VAR} 引用
            env[k] = os.path.expandvars(v)

        self._process = await asyncio.create_subprocess_exec(
            self.config.command, *self.config.args,
            stdin=asyncio.subprocess.PIPE,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
            env=env,
        )

    async def _connect_sse(self) -> None:
        """SSE 传输 — 通过 HTTP SSE 流接收消息，POST 发送请求。"""
        # 延迟实现：需要 httpx-sse 或 aiohttp
        pass

    async def _connect_http(self) -> None:
        """Streamable HTTP 传输 — 标准 HTTP POST。"""
        # 延迟实现：直接用 httpx
        pass

    # ── JSON-RPC 通信 ─────────────────────────────

    async def _send_request(self, method: str, params: Optional[Dict] = None) -> Any:
        """发送 JSON-RPC 请求并等待响应。"""
        self._request_id += 1
        request = {
            "jsonrpc": "2.0",
            "id": self._request_id,
            "method": method,
        }
        if params:
            request["params"] = params

        if self.config.transport == "stdio":
            return await self._stdio_request(request)
        elif self.config.transport == "streamable_http":
            return await self._http_request(request)
        else:
            return await self._sse_request(request)

    async def _stdio_request(self, request: Dict) -> Any:
        """通过 stdin/stdout 发送 JSON-RPC。"""
        assert self._process and self._process.stdin and self._process.stdout

        payload = json.dumps(request) + "\n"
        self._process.stdin.write(payload.encode())
        await self._process.stdin.drain()

        line = await asyncio.wait_for(
            self._process.stdout.readline(), timeout=30.0
        )
        response = json.loads(line.decode())

        if "error" in response:
            raise RuntimeError(f"MCP error: {response['error']}")
        return response.get("result")

    async def _http_request(self, request: Dict) -> Any:
        """通过 HTTP POST 发送 JSON-RPC。"""
        import httpx
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                self.config.url,
                json=request,
                timeout=30.0,
            )
            resp.raise_for_status()
            data = resp.json()
            if "error" in data:
                raise RuntimeError(f"MCP error: {data['error']}")
            return data.get("result")

    async def _sse_request(self, request: Dict) -> Any:
        """通过 SSE 传输发送请求。"""
        # SSE 模式：POST 发送，SSE 流接收
        # 延迟实现
        pass

    # ── MCP 协议 ──────────────────────────────────

    async def _initialize_handshake(self) -> None:
        """MCP initialize + initialized 握手。"""
        await self._send_request("initialize", {
            "protocolVersion": "2024-11-05",
            "capabilities": {},
            "clientInfo": {
                "name": "OmniCore",
                "version": "0.1.0",
            },
        })
        # 发送 initialized 通知（无 id）
        if self.config.transport == "stdio" and self._process and self._process.stdin:
            notif = json.dumps({
                "jsonrpc": "2.0",
                "method": "notifications/initialized",
            }) + "\n"
            self._process.stdin.write(notif.encode())
            await self._process.stdin.drain()

    async def discover_tools(self) -> List[MCPTool]:
        """调用 tools/list 获取可用工具。"""
        result = await self._send_request("tools/list")
        self._tools = []
        for tool_def in result.get("tools", []):
            self._tools.append(MCPTool(
                server_name=self.config.name,
                name=tool_def["name"],
                description=tool_def.get("description", ""),
                input_schema=tool_def.get("inputSchema", {}),
            ))
        log_agent_action(
            f"MCP[{self.config.name}]",
            f"Discovered {len(self._tools)} tools: "
            f"{[t.name for t in self._tools]}",
        )
        return self._tools

    async def call_tool(self, tool_name: str, arguments: Dict[str, Any]) -> Any:
        """调用 tools/call 执行工具。"""
        result = await self._send_request("tools/call", {
            "name": tool_name,
            "arguments": arguments,
        })
        return result

    @property
    def tools(self) -> List[MCPTool]:
        return self._tools


class MCPClientManager:
    """
    管理所有 MCP Server 连接的单例。

    职责：
    1. 从 config/mcp_servers.yaml 加载配置
    2. 启动所有 enabled 的 Server
    3. 汇总所有工具列表
    4. 路由工具调用到对应 Server
    """

    _instance: Optional["MCPClientManager"] = None

    def __init__(self):
        self._clients: Dict[str, MCPClient] = {}
        self._tool_to_server: Dict[str, str] = {}   # tool_name → server_name

    @classmethod
    async def get_instance(cls) -> "MCPClientManager":
        if cls._instance is None:
            cls._instance = cls()
            await cls._instance._load_and_connect()
        return cls._instance

    async def _load_and_connect(self) -> None:
        """从配置文件加载并连接所有 MCP Server。"""
        import yaml
        config_path = os.path.join(
            os.path.dirname(__file__), "..", "config", "mcp_servers.yaml"
        )
        if not os.path.exists(config_path):
            log_warning("MCP config not found, skipping MCP initialization")
            return

        with open(config_path, "r", encoding="utf-8") as f:
            raw = yaml.safe_load(f) or {}

        for name, server_cfg in raw.get("servers", {}).items():
            if not server_cfg.get("enabled", True):
                continue

            config = MCPServerConfig(
                name=name,
                transport=server_cfg.get("transport", "stdio"),
                enabled=True,
                description=server_cfg.get("description", ""),
                description_zh=server_cfg.get("description_zh", ""),
                risk_level=server_cfg.get("risk_level", "medium"),
                max_parallelism=server_cfg.get("max_parallelism", 2),
                command=server_cfg.get("command", ""),
                args=server_cfg.get("args", []),
                env=server_cfg.get("env", {}),
                url=server_cfg.get("url", ""),
            )

            client = MCPClient(config)
            try:
                await client.connect()
                tools = await client.discover_tools()
                self._clients[name] = client

                # 建立工具名 → Server 的映射
                for tool in tools:
                    full_name = f"mcp.{name}.{tool.name}"
                    self._tool_to_server[full_name] = name

            except Exception as e:
                log_error(f"Failed to connect MCP server '{name}': {e}")

    def get_all_tools(self) -> List[MCPTool]:
        """返回所有已连接 Server 的工具列表。"""
        tools = []
        for client in self._clients.values():
            tools.extend(client.tools)
        return tools

    def get_client(self, server_name: str) -> Optional[MCPClient]:
        return self._clients.get(server_name)

    def resolve_server(self, full_tool_name: str) -> Optional[str]:
        """从完整工具名解析出 Server 名。"""
        return self._tool_to_server.get(full_tool_name)

    async def call_tool(self, full_tool_name: str, arguments: Dict[str, Any]) -> Any:
        """路由工具调用到对应 Server。"""
        server_name = self._tool_to_server.get(full_tool_name)
        if not server_name:
            raise ValueError(f"Unknown MCP tool: {full_tool_name}")

        client = self._clients[server_name]
        # 去掉 mcp.{server_name}. 前缀，得到原始 tool name
        original_name = full_tool_name.split(".", 2)[2]
        return await client.call_tool(original_name, arguments)

    async def shutdown(self) -> None:
        """关闭所有连接。"""
        for client in self._clients.values():
            await client.disconnect()
        self._clients.clear()
        self._tool_to_server.clear()
```

```python
@tool_adapter("mcp_adapter")
class MCPToolAdapter(BaseToolAdapter):
    """MCP 协议工具适配器 — 将 TaskItem 转发到 MCP Server。"""

    async def execute(
        self,
        task: Dict[str, Any],
        shared_memory_snapshot: Dict[str, Any],
        registered_tool: RegisteredTool,
    ) -> Dict[str, Any]:
        from core.mcp_client import MCPClientManager

        full_tool_name = task.get("tool_name", "")
        tool_args = task.get("params", {}) or task.get("tool_args", {})

        try:
            manager = await MCPClientManager.get_instance()
            result = await manager.call_tool(full_tool_name, tool_args)

            # MCP 返回的 content 数组 → 提取文本
            output_text = ""
            if isinstance(result, dict) and "content" in result:
                for item in result["content"]:
                    if item.get("type") == "text":
                        output_text += item.get("text", "")
            else:
                output_text = str(result)

            return _base_outcome(
                task,
                registered_tool,
                status=str(TaskStatus.COMPLETED),
                result={"success": True, "output": output_text, "raw": result},
                shared_memory=output_text,
                error_trace="",
                failure_type=None,
            )

        except Exception as e:
            log_error(f"MCP tool execution failed: {e}")
            return _base_outcome(
                task,
                registered_tool,
                status=str(TaskStatus.FAILED),
                result={"success": False, "error": str(e)},
                shared_memory=None,
                error_trace=str(e),
                failure_type=str(FailureType.UNKNOWN),
            )
```

### 3.4 动态工具注册

在 `core/tool_registry.py` 的 `_register_builtin_tools()` 末尾追加：

```python
def _register_mcp_tools(registry: ToolRegistry) -> None:
    """从 MCPClientManager 动态注册所有已发现的 MCP 工具。"""
    import asyncio
    from core.mcp_client import MCPClientManager

    try:
        loop = asyncio.get_event_loop()
        if loop.is_running():
            # 在已有事件循环中（如 Streamlit），延迟注册
            return
        manager = loop.run_until_complete(MCPClientManager.get_instance())
    except Exception:
        return

    for tool in manager.get_all_tools():
        full_name = f"mcp.{tool.server_name}.{tool.name}"
        server_name = tool.server_name

        # 从 mcp_servers.yaml 获取 Server 级别的配置
        client = manager.get_client(server_name)
        if not client:
            continue

        registry.register(
            RegisteredTool(
                spec=ToolSpec(
                    name=full_name,
                    task_type="mcp_handler",
                    description=tool.description or f"MCP tool: {tool.name}",
                    risk_level=client.config.risk_level,
                    tags=["mcp", server_name],
                    input_schema=tool.input_schema,
                ),
                adapter_name="mcp_adapter",
                max_parallelism=client.config.max_parallelism,
            )
        )
```

### 3.5 Router 自动感知

无需额外修改。流程：

1. `_register_mcp_tools()` 将 MCP 工具注册到 `ToolRegistry`
2. `build_dynamic_tool_prompt_lines()` 遍历 `registry.list_tools()` 时自动包含 MCP 工具
3. Router prompt 自动获得类似如下内容：

```
- mcp.github.create_issue (adapter: mcp_adapter, task_type: mcp_handler, risk: medium) - Create a new issue in a GitHub repository
- mcp.filesystem.read_file (adapter: mcp_adapter, task_type: mcp_handler, risk: medium) - Read the contents of a file
```

4. Router LLM 正常选择 `tool_name: "mcp.github.create_issue"` → TaskExecutor 分发到 `MCPToolAdapter`

## 4. 策略引擎集成

在 `core/policy_engine.py` 中为 MCP 工具增加策略规则：

```python
# MCP 工具风险策略
MCP_HIGH_RISK_ACTIONS = [
    "delete", "remove", "drop", "send_message", "post",
    "create_issue", "merge", "close",
]

def _evaluate_mcp_task(task: Dict) -> PolicyDecision:
    tool_name = task.get("tool_name", "")
    description = task.get("description", "").lower()

    # 从 mcp_servers.yaml 继承 Server 级别的 risk_level
    if any(action in tool_name.lower() for action in MCP_HIGH_RISK_ACTIONS):
        return PolicyDecision(
            approved=False,
            requires_confirmation=True,
            reason=f"MCP 工具 {tool_name} 包含高风险操作，需要人工确认",
        )

    # read-only 操作自动放行
    return PolicyDecision(approved=True, requires_confirmation=False)
```

## 7. 风险与注意事项

| 风险 | 缓解措施 |
|------|----------|
| MCP Server 启动慢，影响 OmniCore 启动 | 懒加载：首次使用 MCP 工具时才连接 |
| MCP Server 崩溃 | 自动重连机制 + 任务降级为 FAILED |
| 工具名冲突（多个 Server 有同名工具） | 使用 `mcp.{server_name}.{tool_name}` 全限定名 |
| stdio 模式下子进程泄露 | `MCPClientManager.shutdown()` + atexit 注册 |
| 环境变量包含密钥 | `mcp_servers.yaml` 支持 `${ENV_VAR}` 引用，不硬编码密钥 |
