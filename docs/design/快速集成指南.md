```python
async def enhanced_web_worker_adapter(task: Dict[str, Any], state: OmniCoreState) -> Dict[str, Any]:
    """增强版 Web Worker 适配器"""
    from agents.enhanced_web_worker import EnhancedWebWorker
    from utils.browser_toolkit import BrowserToolkit

    worker = EnhancedWebWorker()
    task_desc = task.get("description", "")
    limit = task.get("params", {}).get("limit", 10)
    headless = task.get("params", {}).get("headless", True)

    async with BrowserToolkit(headless=headless) as toolkit:
        # 如果有 URL，先导航
        url = task.get("params", {}).get("url")
        if url:
            goto_r = await toolkit.goto(url)
            if not goto_r.success:
                return {
                    "success": False,
                    "error": f"导航失败: {goto_r.error}",
                    "data": []
                }
            await toolkit.human_delay(2000, 3000)

        # 执行智能提取
        result = await worker.smart_extract(
            toolkit=toolkit,
            task_description=task_desc,
            limit=limit
        )

        return result
```

## 方案 C：为 Browser Agent 增强感知

在 `agents/browser_agent.py` 中增强决策能力：

### 在 BrowserAgent 类中添加方法

```python
async def _perceive_and_decide(self, task: str) -> BrowserAction:
    """基于页面感知做决策"""
    from utils.page_perceiver import get_page_understanding

    # 获取页面理解
    understanding = await get_page_understanding(self.toolkit, task)

    # 提取可交互元素
    elements = await self._extract_interactive_elements()

    # 构建决策 prompt
    prompt = f"""你是浏览器操作规划专家。基于页面理解和任务目标，决定下一步操作。

## 任务
{task}

## 页面理解
{understanding}

## 可交互元素
{self._format_elements_for_llm(elements[:20])}

## 可用操作
- click: 点击元素
- input: 输入文本
- scroll: 滚动页面
- extract: 提取数据
- done: 完成任务

返回 JSON：
{{
    "action": "click",
    "element_index": 5,
    "value": "",
    "reasoning": "需要点击搜索按钮"
}}
"""

    response = self._get_llm().chat_with_system(
        system_prompt=prompt,
        user_message="请决定下一步操作",
        temperature=0.3,
        json_mode=True
    )

    decision = self._get_llm().parse_json_response(response)

    # 转换为 BrowserAction
    action_type_map = {
        "click": ActionType.CLICK,
        "input": ActionType.INPUT,
        "scroll": ActionType.SCROLL,
        "extract": ActionType.EXTRACT,
        "done": ActionType.DONE,
    }

    return BrowserAction(
        action_type=action_type_map.get(decision.get("action"), ActionType.WAIT),
        target_selector=elements[decision.get("element_index", 0)].selector if elements else "",
        value=decision.get("value", ""),
        description=decision.get("reasoning", ""),
        confidence=0.8
    )
```

## 常见问题

### Q1: 页面理解返回空结果？
**A**: 检查页面是否加载完成，增加等待时间：
```python
await toolkit.human_delay(3000, 5000)
```

### Q2: 选择器生成不准确？
**A**: 调整 prompt，增加更多示例和规则。或者在 `page_perceiver.py` 中调整 JS 提取逻辑。

### Q3: 性能太慢？
**A**:
1. 启用缓存（已默认启用）
2. 减少提取的元素数量（在 JS 中调整 `slice(0, 60)` 等限制）
3. 使用更快的模型（如 Claude Haiku）

### Q4: 如何回退到旧逻辑？
**A**: 在方法开头添加开关：
```python
async def analyze_page_structure(self, tk, task):
    USE_ENHANCED = True  # 设为 False 回退

    if USE_ENHANCED:
        # 新逻辑
        return await self._enhanced_analyze(tk, task)
    else:
        # 旧逻辑
        return await self._legacy_analyze(tk, task)
```
