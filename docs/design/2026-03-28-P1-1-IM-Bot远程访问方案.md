# P1-1: IM Bot 远程访问方案

> 优先级：P1 | 预估工作量：2-3 天 | 依赖：无

---

## 1. 目标

通过 IM Bot（Telegram 优先，后续可扩展微信/飞书）为 OmniCore 增加远程访问能力，实现手机端任务下发、进度跟踪、结果接收、审批操作。

**验收标准**：
- 通过 Telegram 发送消息即可创建 OmniCore 任务
- 任务完成后自动推送结果（文本 + 文件/截图）
- 支持 `/status`、`/cancel`、`/approve`、`/reject` 交互命令
- 高风险操作的审批通过 Telegram 内联按钮完成

---

## 2. 现有代码接入点

| 组件 | 文件 | 说明 |
|------|------|------|
| 任务提交 | `core/runtime.py` `submit_task()` | 接受 `user_input` + `trigger_source` 参数 |
| 队列管理 | `utils/runtime_state_store.py` `submit_job()` | Job 记录包含 `trigger_source` 字段 |
| Worker 消费 | `core/runtime.py` `run_next_queued_task()` | 从队列取任务执行 |
| 任务完成 | `utils/runtime_state_store.py` `complete_job()` | 更新状态 + 写入通知 |
| 通知系统 | `utils/runtime_state_store.py` `create_notification()` | 支持按 `trigger_source` 过滤 |
| 审批机制 | `core/policy_engine.py` | `waiting_for_approval` 状态 + `approve/reject` |
| 定时任务 | `core/runtime.py` `create_scheduled_task()` | 可复用定时推送 |

**关键 API 签名**：

```python
# 任务提交
async def submit_task(
    user_input: str,
    session_id: str = None,
    trigger_source: str = "cli",     # ← 改为 "telegram_bot"
    priority: int = 10,
) -> Dict[str, Any]:
    # 返回 {"job_id": "...", "status": "queued"}

# Job 完成后的状态
state_store.complete_job(
    job_id=job_id,
    status="completed",
    output=output_text,
    artifacts=artifacts_list,
)
```

## 4. 详细设计

### 4.1 Telegram Bot 模块

新建 `interactions/telegram_bot.py`：

```python
"""
Telegram Bot — OmniCore 远程交互入口。

依赖：python-telegram-bot >= 20.0
"""

import asyncio
import json
from typing import Dict, Optional
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import (
    Application, CommandHandler, MessageHandler,
    CallbackQueryHandler, filters,
)

from utils.logger import log_agent_action, log_error


class TelegramBotServer:
    """Telegram Bot 服务端，桥接 IM 消息与 OmniCore Runtime。"""

    def __init__(self, token: str, allowed_user_ids: list[int]):
        self.token = token
        self.allowed_user_ids = set(allowed_user_ids)
        self._job_mapping: Dict[str, Dict] = {}  # job_id → {chat_id, message_id}
        self._app: Optional[Application] = None

    # ── 启动与关闭 ────────────────────────────

    async def start(self) -> None:
        """启动 Bot 并开始监听。"""
        self._app = Application.builder().token(self.token).build()

        # 注册命令
        self._app.add_handler(CommandHandler("start", self._cmd_start))
        self._app.add_handler(CommandHandler("status", self._cmd_status))
        self._app.add_handler(CommandHandler("cancel", self._cmd_cancel))
        self._app.add_handler(CommandHandler("jobs", self._cmd_jobs))
        self._app.add_handler(CommandHandler("skills", self._cmd_skills))

        # 审批回调
        self._app.add_handler(CallbackQueryHandler(self._callback_approval))

        # 普通消息 → 创建任务
        self._app.add_handler(
            MessageHandler(filters.TEXT & ~filters.COMMAND, self._on_message)
        )

        await self._app.initialize()
        await self._app.start()
        await self._app.updater.start_polling()
        log_agent_action("TelegramBot", "Bot started, listening for messages")

    async def stop(self) -> None:
        if self._app:
            await self._app.updater.stop()
            await self._app.stop()
            await self._app.shutdown()

    # ── 权限检查 ──────────────────────────────

    def _is_authorized(self, update: Update) -> bool:
        user_id = update.effective_user.id
        return user_id in self.allowed_user_ids

    # ── 消息处理 ──────────────────────────────

    async def _on_message(self, update: Update, context) -> None:
        """普通文本消息 → 创建 OmniCore 任务。"""
        if not self._is_authorized(update):
            await update.message.reply_text("⛔ 未授权用户")
            return

        user_input = update.message.text.strip()
        if not user_input:
            return

        chat_id = update.effective_chat.id

        # 提交任务
        from core.runtime import submit_task
        result = await submit_task(
            user_input=user_input,
            trigger_source="telegram_bot",
            metadata={"chat_id": chat_id, "user_id": update.effective_user.id},
        )

        job_id = result.get("job_id", "unknown")
        self._job_mapping[job_id] = {
            "chat_id": chat_id,
            "message_id": update.message.message_id,
        }

        await update.message.reply_text(
            f"✅ 任务已提交\n"
            f"Job ID: `{job_id}`\n"
            f"状态: queued\n\n"
            f"完成后会自动通知你。",
            parse_mode="Markdown",
        )

    # ── 命令处理 ──────────────────────────────

    async def _cmd_start(self, update: Update, context) -> None:
        await update.message.reply_text(
            "🤖 OmniCore Agent Bot\n\n"
            "直接发送文字即可创建任务。\n\n"
            "命令列表：\n"
            "/status <job_id> — 查看任务状态\n"
            "/cancel <job_id> — 取消任务\n"
            "/jobs — 查看最近任务\n"
            "/skills — 查看已有技能\n"
        )

    async def _cmd_status(self, update: Update, context) -> None:
        if not self._is_authorized(update):
            return

        args = context.args
        if not args:
            await update.message.reply_text("用法: /status <job_id>")
            return

        from utils.runtime_state_store import get_state_store
        store = get_state_store()
        job = store.get_job(args[0])
        if not job:
            await update.message.reply_text("未找到该任务")
            return

        status_emoji = {
            "queued": "⏳", "executing": "🔄",
            "completed": "✅", "failed": "❌",
            "waiting_for_approval": "⏸",
        }
        emoji = status_emoji.get(job.get("status", ""), "❓")

        await update.message.reply_text(
            f"{emoji} Job: `{args[0]}`\n"
            f"状态: {job.get('status')}\n"
            f"创建: {job.get('created_at', 'N/A')}\n"
            f"输出: {(job.get('output', '') or 'N/A')[:500]}",
            parse_mode="Markdown",
        )

    async def _cmd_cancel(self, update: Update, context) -> None:
        if not self._is_authorized(update):
            return
        args = context.args
        if not args:
            await update.message.reply_text("用法: /cancel <job_id>")
            return

        from utils.runtime_state_store import get_state_store
        store = get_state_store()
        store.update_job_status(args[0], "cancelled")
        await update.message.reply_text(f"✅ 已取消任务 `{args[0]}`", parse_mode="Markdown")

    async def _cmd_jobs(self, update: Update, context) -> None:
        if not self._is_authorized(update):
            return

        from utils.runtime_state_store import get_state_store
        store = get_state_store()
        jobs = store.list_recent_jobs(limit=10)

        if not jobs:
            await update.message.reply_text("暂无任务")
            return

        lines = []
        for job in jobs:
            status = job.get("status", "unknown")
            desc = (job.get("user_input", "") or "")[:50]
            lines.append(f"• `{job['job_id'][:8]}` [{status}] {desc}")

        await update.message.reply_text("\n".join(lines), parse_mode="Markdown")

    async def _cmd_skills(self, update: Update, context) -> None:
        if not self._is_authorized(update):
            return

        try:
            from memory.skill_store import SkillStore
            store = SkillStore()
            skills = store.list_skills()
            if not skills:
                await update.message.reply_text("暂无已学习的技能")
                return

            lines = []
            for s in skills:
                total = s.success_count + s.failure_count
                rate = s.success_count / max(total, 1)
                lines.append(f"• {s.name} (成功率={rate:.0%}, 使用={total}次)")
            await update.message.reply_text("\n".join(lines))
        except ImportError:
            await update.message.reply_text("Skill Library 尚未启用")

    # ── 审批回调 ──────────────────────────────

    async def _callback_approval(self, update: Update, context) -> None:
        """处理内联按钮的审批回调。"""
        query = update.callback_query
        await query.answer()

        data = query.data  # 格式: "approve:{job_id}" 或 "reject:{job_id}"
        action, job_id = data.split(":", 1)

        from core.runtime import handle_approval
        if action == "approve":
            await handle_approval(job_id, approved=True)
            await query.edit_message_text(f"✅ 已批准任务 `{job_id}`", parse_mode="Markdown")
        elif action == "reject":
            await handle_approval(job_id, approved=False)
            await query.edit_message_text(f"❌ 已拒绝任务 `{job_id}`", parse_mode="Markdown")

    # ── 主动推送 ──────────────────────────────

    async def push_job_result(self, job_id: str, result: Dict) -> None:
        """任务完成后主动推送结果到 Telegram。"""
        mapping = self._job_mapping.get(job_id)
        if not mapping or not self._app:
            return

        chat_id = mapping["chat_id"]
        status = result.get("status", "unknown")
        output = (result.get("output", "") or "")[:3000]

        status_emoji = {"completed": "✅", "failed": "❌"}
        emoji = status_emoji.get(status, "❓")

        await self._app.bot.send_message(
            chat_id=chat_id,
            text=f"{emoji} 任务完成\n\nJob: `{job_id}`\n\n{output}",
            parse_mode="Markdown",
        )

        # 推送制品文件（截图、下载等）
        artifacts = result.get("artifacts", [])
        for artifact in artifacts[:5]:  # 最多推送 5 个
            path = artifact.get("path", "")
            if path and path.endswith((".png", ".jpg", ".jpeg")):
                try:
                    await self._app.bot.send_photo(chat_id=chat_id, photo=open(path, "rb"))
                except Exception as e:
                    log_error(f"Failed to send photo: {e}")

    async def push_approval_request(self, job_id: str, reason: str) -> None:
        """高风险操作时推送审批请求。"""
        mapping = self._job_mapping.get(job_id)
        if not mapping or not self._app:
            return

        keyboard = InlineKeyboardMarkup([
            [
                InlineKeyboardButton("✅ 批准", callback_data=f"approve:{job_id}"),
                InlineKeyboardButton("❌ 拒绝", callback_data=f"reject:{job_id}"),
            ]
        ])

        await self._app.bot.send_message(
            chat_id=mapping["chat_id"],
            text=f"⚠️ 需要审批\n\nJob: `{job_id}`\n原因: {reason}",
            reply_markup=keyboard,
            parse_mode="Markdown",
        )
```

### 4.2 通知路由器

新建 `interactions/notification_router.py`：

```python
"""
NotificationRouter — 按 trigger_source 将通知路由到对应渠道。
"""

class NotificationRouter:
    """将任务完成/审批通知路由到对应的交互渠道。"""

    _handlers = {}

    @classmethod
    def register(cls, trigger_source: str, handler):
        cls._handlers[trigger_source] = handler

    @classmethod
    async def notify_job_completed(cls, job_id: str, result: Dict, trigger_source: str):
        handler = cls._handlers.get(trigger_source)
        if handler and hasattr(handler, "push_job_result"):
            await handler.push_job_result(job_id, result)

    @classmethod
    async def notify_approval_needed(cls, job_id: str, reason: str, trigger_source: str):
        handler = cls._handlers.get(trigger_source)
        if handler and hasattr(handler, "push_approval_request"):
            await handler.push_approval_request(job_id, reason)
```

## 7. 风险与注意事项

| 风险 | 缓解措施 |
|------|----------|
| Bot Token 泄露 | 只存 `.env`，不提交；`allowed_user_ids` 白名单 |
| 并发消息洪泛 | 任务队列自带限流；Bot 端加消息去抖（3s） |
| 大文件推送失败 | Telegram 文件限制 50MB；超限只推文本摘要 |
| 长时间任务无反馈 | 增加进度中间推送（Worker 心跳） |
| 网络不稳定 | python-telegram-bot 自带重连机制 |
