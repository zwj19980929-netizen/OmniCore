# P3-1: 实时信息流 & 事件驱动方案

> 优先级：P3 | 预估工作量：3-5 天 | 依赖：P1-1 IM Bot（通知推送）

---

## 1. 目标

将 OmniCore 从"被动响应"升级为"主动监控"：网页内容变化、邮件到达、Webhook 触发时，自动创建任务并执行，结果通过 IM Bot 或通知系统推送给用户。

**验收标准**：
- `/watch url <url> [条件]` — 监控网页变化，变化时触发任务
- `/watch email [关键词]` — 监控邮件，匹配时触发任务
- Webhook endpoint 接收外部事件触发任务
- 所有事件源统一抽象为 `EventSource` 接口

---

### 4.1 EventSource 协议

新建 `utils/event_sources/base.py`：

```python
"""
EventSource 协议 — 所有事件源的统一接口。
"""

from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional, Protocol


@dataclass
class WatchEvent:
    """统一的事件格式。"""
    event_id: str
    source_type: str                        # "web_page" | "email" | "webhook" | "directory" | "schedule"
    source_id: str                          # Watch 配置的 ID
    session_id: str
    trigger_data: Dict[str, Any]            # 触发事件的原始数据
    user_input_template: str                # 任务描述模板，支持 {变量} 替换
    goal_id: str = ""
    project_id: str = ""
    todo_id: str = ""
    created_at: str = ""


class EventSource(Protocol):
    """事件源协议，所有实现必须满足此接口。"""

    def get_source_type(self) -> str:
        """返回事件源类型标识。"""
        ...

    def poll_events(self, limit: int = 10) -> List[WatchEvent]:
        """
        轮询并返回新事件列表。
        实现应保证幂等：同一事件不能返回两次。
        """
        ...

    def create_watch(self, config: Dict[str, Any]) -> str:
        """创建监控配置，返回 watch_id。"""
        ...

    def pause_watch(self, watch_id: str) -> None:
        """暂停监控。"""
        ...

    def delete_watch(self, watch_id: str) -> None:
        """删除监控。"""
        ...

    def list_watches(self) -> List[Dict[str, Any]]:
        """列出所有监控配置。"""
        ...
```

### 4.2 网页变化监控

新建 `utils/event_sources/web_page_watch.py`：

```python
"""
WebPageWatchSource — 定期抓取网页并检测内容变化。

实现策略：
1. 用 WebWorker 抓取目标页面文本
2. 与上次快照做文本 diff（Levenshtein 距离 / hash 比较）
3. 变化超过阈值时生成 WatchEvent
"""

import hashlib
import json
import os
from datetime import datetime
from typing import Any, Dict, List, Optional

from utils.logger import log_agent_action, log_warning


class WebPageWatchSource:
    """网页内容变化监控。"""

    def __init__(self, data_dir: str = "data"):
        self._watch_file = os.path.join(data_dir, "web_page_watches.jsonl")
        self._snapshot_dir = os.path.join(data_dir, "web_snapshots")
        os.makedirs(self._snapshot_dir, exist_ok=True)

    def get_source_type(self) -> str:
        return "web_page"

    def create_watch(self, config: Dict[str, Any]) -> str:
        """
        创建网页监控。

        config:
            url: 要监控的 URL
            session_id: 会话 ID
            check_interval_seconds: 检查间隔（默认 3600 = 1小时）
            change_threshold: 变化阈值（0~1，默认 0.1 = 10% 变化触发）
            user_input_template: 触发时的任务描述
            goal_id / project_id / todo_id: 可选上下文
            extract_selector: CSS 选择器，只监控页面特定区域（可选）
            note: 备注
        """
        import uuid
        watch_id = f"web_watch_{uuid.uuid4().hex[:12]}"

        watch = {
            "watch_id": watch_id,
            "url": config["url"],
            "session_id": config.get("session_id", ""),
            "check_interval_seconds": config.get("check_interval_seconds", 3600),
            "change_threshold": config.get("change_threshold", 0.1),
            "user_input_template": config.get(
                "user_input_template",
                "网页 {url} 内容发生了变化，请分析变化内容并汇报。",
            ),
            "extract_selector": config.get("extract_selector", ""),
            "goal_id": config.get("goal_id", ""),
            "project_id": config.get("project_id", ""),
            "todo_id": config.get("todo_id", ""),
            "note": config.get("note", ""),
            "status": "active",
            "last_snapshot_hash": "",
            "last_checked_at": "",
            "next_check_at": datetime.now().isoformat(),
            "created_at": datetime.now().isoformat(),
        }

        self._append_watch(watch)
        log_agent_action("WebPageWatch", f"Created watch: {watch_id} → {config['url']}")
        return watch_id

    def poll_events(self, limit: int = 5) -> List:
        """检查所有到期的网页监控，返回变化事件。"""
        from utils.event_sources.base import WatchEvent
        import uuid

        watches = self._load_watches()
        now = datetime.now()
        events = []
        updated_watches = []

        for watch in watches:
            if watch.get("status") != "active":
                updated_watches.append(watch)
                continue

            next_check = watch.get("next_check_at", "")
            if next_check and datetime.fromisoformat(next_check) > now:
                updated_watches.append(watch)
                continue

            if len(events) >= limit:
                updated_watches.append(watch)
                continue

            # 执行检查
            changed, new_hash, new_content = self._check_page(watch)

            if changed:
                url = watch["url"]
                events.append(WatchEvent(
                    event_id=f"event_{uuid.uuid4().hex[:12]}",
                    source_type="web_page",
                    source_id=watch["watch_id"],
                    session_id=watch["session_id"],
                    trigger_data={
                        "url": url,
                        "new_hash": new_hash,
                    },
                    user_input_template=watch.get(
                        "user_input_template",
                        f"网页 {url} 内容发生了变化，请分析。",
                    ).format(url=url),
                    goal_id=watch.get("goal_id", ""),
                    project_id=watch.get("project_id", ""),
                    todo_id=watch.get("todo_id", ""),
                    created_at=now.isoformat(),
                ))

            # 更新检查时间
            interval = watch.get("check_interval_seconds", 3600)
            from datetime import timedelta
            watch["last_checked_at"] = now.isoformat()
            watch["next_check_at"] = (now + timedelta(seconds=interval)).isoformat()
            watch["last_snapshot_hash"] = new_hash
            updated_watches.append(watch)

        self._save_watches(updated_watches)
        return events

    def _check_page(self, watch: Dict) -> tuple:
        """抓取页面并与上次快照比较。返回 (changed, new_hash, new_content)。"""
        url = watch["url"]
        try:
            # 用 requests 轻量抓取（优先 static fetch）
            import requests
            resp = requests.get(url, timeout=15, headers={
                "User-Agent": "Mozilla/5.0 (compatible; OmniCore/1.0)"
            })
            resp.raise_for_status()

            # 提取纯文本
            from bs4 import BeautifulSoup
            soup = BeautifulSoup(resp.text, "html.parser")

            selector = watch.get("extract_selector", "")
            if selector:
                el = soup.select_one(selector)
                text = el.get_text(strip=True) if el else soup.get_text(strip=True)
            else:
                # 去掉 script/style，取正文
                for tag in soup(["script", "style", "nav", "footer"]):
                    tag.decompose()
                text = soup.get_text(separator=" ", strip=True)

            new_hash = hashlib.md5(text.encode()).hexdigest()
            old_hash = watch.get("last_snapshot_hash", "")

            if not old_hash:
                # 第一次检查，只记录快照不触发
                return False, new_hash, text

            if new_hash == old_hash:
                return False, new_hash, text

            # hash 不同，做文本相似度判断
            threshold = watch.get("change_threshold", 0.1)
            change_ratio = self._compute_change_ratio(old_hash, new_hash, watch, text)
            changed = change_ratio >= threshold

            return changed, new_hash, text

        except Exception as e:
            log_warning(f"WebPageWatch check failed for {url}: {e}")
            return False, watch.get("last_snapshot_hash", ""), ""

    def _compute_change_ratio(
        self, old_hash: str, new_hash: str, watch: Dict, new_content: str
    ) -> float:
        """粗略估算内容变化比例。"""
        # 简化实现：hash 不同则认为有变化，返回固定比例
        # 精确实现可以保存上次 snapshot 文件做 diff
        snapshot_path = os.path.join(
            self._snapshot_dir, f"{watch['watch_id']}.txt"
        )
        if os.path.exists(snapshot_path):
            with open(snapshot_path, "r", encoding="utf-8") as f:
                old_content = f.read()
            # 简单的行级 diff 估算
            old_lines = set(old_content.split())
            new_lines = set(new_content.split())
            if not old_lines:
                return 1.0
            changed = len(old_lines.symmetric_difference(new_lines))
            total = max(len(old_lines), len(new_lines))
            return changed / total if total > 0 else 0.0

        # 没有快照文件，认为 hash 变化即触发
        with open(snapshot_path, "w", encoding="utf-8") as f:
            f.write(new_content)
        return 1.0

    def _load_watches(self) -> List[Dict]:
        if not os.path.exists(self._watch_file):
            return []
        watches = []
        with open(self._watch_file, "r", encoding="utf-8") as f:
            for line in f:
                try:
                    watches.append(json.loads(line.strip()))
                except Exception:
                    pass
        return watches

    def _append_watch(self, watch: Dict) -> None:
        with open(self._watch_file, "a", encoding="utf-8") as f:
            f.write(json.dumps(watch, ensure_ascii=False) + "\n")

    def _save_watches(self, watches: List[Dict]) -> None:
        with open(self._watch_file, "w", encoding="utf-8") as f:
            for w in watches:
                f.write(json.dumps(w, ensure_ascii=False) + "\n")

    def list_watches(self) -> List[Dict]:
        return [w for w in self._load_watches() if w.get("status") == "active"]

    def pause_watch(self, watch_id: str) -> None:
        watches = self._load_watches()
        for w in watches:
            if w["watch_id"] == watch_id:
                w["status"] = "paused"
        self._save_watches(watches)

    def delete_watch(self, watch_id: str) -> None:
        watches = [w for w in self._load_watches() if w["watch_id"] != watch_id]
        self._save_watches(watches)
```

### 4.4 Webhook 接收器

```python
"""
WebhookSource — HTTP Webhook 事件接收。
启动一个轻量 HTTP 服务器监听 webhook 调用。
"""

import json
import queue
import threading
from http.server import BaseHTTPRequestHandler, HTTPServer
from typing import Any, Dict, List

from utils.logger import log_agent_action


_event_queue: queue.Queue = queue.Queue()


class WebhookHandler(BaseHTTPRequestHandler):
    def do_POST(self):
        content_length = int(self.headers.get("Content-Length", 0))
        body = self.rfile.read(content_length)
        try:
            payload = json.loads(body)
        except Exception:
            payload = {"raw": body.decode(errors="replace")}

        # 从路径解析 watch_id: /webhook/{watch_id}
        watch_id = self.path.lstrip("/webhook/").strip("/")
        _event_queue.put({"watch_id": watch_id, "payload": payload})

        self.send_response(200)
        self.end_headers()
        self.wfile.write(b'{"ok": true}')

    def log_message(self, *args):
        pass  # 静默 HTTP 日志


class WebhookSource:
    """Webhook 事件源。"""

    def __init__(self, port: int = 9988):
        self.port = port
        self._server: HTTPServer = None
        self._thread: threading.Thread = None
        self._watches: Dict[str, Dict] = {}

    def start(self) -> None:
        self._server = HTTPServer(("0.0.0.0", self.port), WebhookHandler)
        self._thread = threading.Thread(
            target=self._server.serve_forever, daemon=True
        )
        self._thread.start()
        log_agent_action("WebhookSource", f"Listening on port {self.port}")

    def stop(self) -> None:
        if self._server:
            self._server.shutdown()

    def get_source_type(self) -> str:
        return "webhook"

    def create_watch(self, config: Dict[str, Any]) -> str:
        import uuid
        watch_id = f"webhook_{uuid.uuid4().hex[:12]}"
        self._watches[watch_id] = config
        return watch_id

    def poll_events(self, limit: int = 10) -> List:
        from utils.event_sources.base import WatchEvent
        import uuid

        events = []
        while len(events) < limit:
            try:
                item = _event_queue.get_nowait()
            except queue.Empty:
                break

            watch_id = item.get("watch_id", "")
            watch_config = self._watches.get(watch_id, {})
            payload = item.get("payload", {})

            template = watch_config.get(
                "user_input_template",
                f"收到 Webhook 事件，内容: {str(payload)[:200]}",
            )

            events.append(WatchEvent(
                event_id=f"wh_{uuid.uuid4().hex[:8]}",
                source_type="webhook",
                source_id=watch_id,
                session_id=watch_config.get("session_id", ""),
                trigger_data=payload,
                user_input_template=template,
            ))

        return events

    def list_watches(self) -> List[Dict]:
        return [{"watch_id": k, **v} for k, v in self._watches.items()]

    def pause_watch(self, watch_id: str) -> None:
        if watch_id in self._watches:
            self._watches[watch_id]["status"] = "paused"

    def delete_watch(self, watch_id: str) -> None:
        self._watches.pop(watch_id, None)
```

### 4.5 EventDispatcher 集成到 Worker 循环

新建 `utils/event_dispatcher.py`：

```python
"""
EventDispatcher — 统一轮询所有 EventSource 并提交任务。
在 Worker 循环中调用，替代现有的多个 _release_*() 函数。
"""

from typing import List
from utils.event_sources.base import WatchEvent
from utils.logger import log_agent_action


class EventDispatcher:
    """统一事件分发器。"""

    def __init__(self):
        self._sources = []

    def register_source(self, source) -> None:
        self._sources.append(source)

    def dispatch_pending_events(self, limit_per_source: int = 5) -> List[str]:
        """轮询所有事件源，返回已提交的 job_id 列表。"""
        from core.runtime import submit_task

        job_ids = []
        for source in self._sources:
            try:
                events = source.poll_events(limit=limit_per_source)
                for event in events:
                    result = submit_task(
                        user_input=event.user_input_template,
                        session_id=event.session_id or None,
                        trigger_source=f"event_{event.source_type}",
                        goal_id=event.goal_id or None,
                        project_id=event.project_id or None,
                        todo_id=event.todo_id or None,
                    )
                    job_id = result.get("job_id", "")
                    if job_id:
                        job_ids.append(job_id)
                        log_agent_action(
                            "EventDispatcher",
                            f"[{event.source_type}] event → job {job_id}",
                        )
            except Exception as e:
                from utils.logger import log_warning
                log_warning(f"EventSource {source.get_source_type()} dispatch failed: {e}")

        return job_ids
```

## 6. 风险与注意事项

| 风险 | 缓解措施 |
|------|----------|
| 网页检查频率过高消耗资源 | 最小间隔 300s（5 分钟），默认 3600s（1 小时）|
| 误触发（动态广告/日期变化） | 提供 `extract_selector` 只监控关键区域；`change_threshold` 可调 |
| IMAP 密码安全 | 引用 env 变量，不存明文 |
| Webhook 端口占用 | `WEBHOOK_PORT` 可配置，默认 9988 |
| 事件风暴（大量事件同时触发）| `limit_per_source=5` 限流 + 队列缓冲 |
